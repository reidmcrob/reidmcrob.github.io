                function multiply(){
                var str = ""
                for (var i = 1; i <= 100; i += 1) {
                  if (i % 3 === 0 && i % 5 === 0) { //if i goes into 3 and 5 evenly
                    str = str + "FooBar" + "";
                  } else if (i % 3 === 0) { //if i is evenly divisible by 3
                    str = str + "Foo" + "";
                  } else if (i % 5 === 0) { //if i is evenly divisible by 5
                    str = str + "Bar" + "";
                  } else { //if not print out the number
                    str = str + i + ""
                  }
                  $("#q2result").html(str);
                }
              }

                $("#q2button").on("click", function() {
                  multiply();
                });
              
                function chessBoard() {
                var size = prompt("Please enter a number to create a length.");
                var line = '';
                var print = "";
                var i = 0;

                for (i = 0; i < size / 2; i++) {
                  line = print + line + '# ';
                }

                for (i = 0; i < size; i++) {
                  if (i % 2 === 0) {
                    print = print + (' ' + line) + "";
                  } else {
                    print = print + (line) + "";
                  }
                }
                $("#q3result").html(print);
              }


              $("#q3button").on("click", function() {

                chessBoard();
              });
                
                function countBs(bString) {
                        var count = 0;
                        for (var i = 0; i < bString.length; i++) {
                          if (bString.charAt(i) == 'B') {
                            count = (count + 1);
                          }
                        }
                        return count;
                      }


                      $("#q4button").on("click", function() {

                        var userInput = prompt("Write several -B- and I will tell you how many there are.");
                        var x = countBs(userInput);

                        $("#q4result").html(x);
                      });
                
                function countChar(cString, charMatch) {
                  var count = 0;
                  for (var i = 0; i < cString.length; i++) {
                    if (cString.charAt(i) == charMatch) {
                      count = (count + 1);
                    }
                  }
                  return count;
                }
