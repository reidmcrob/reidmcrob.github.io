                function multiply(){
                var str = ""
                for (var i = 1; i <= 100; i += 1) {
                  if (i % 3 === 0 && i % 5 === 0) { //if i goes into 3 and 5 evenly
                    str = str + "FooBar" + "";
                  } else if (i % 3 === 0) { //if i is evenly divisible by 3
                    str = str + "Foo" + "";
                  } else if (i % 5 === 0) { //if i is evenly divisible by 5
                    str = str + "Bar" + "";
                  } else { //if not print out the number
                    str = str + i + ""
                  }
                  $("#q2result").html(str);
                }
              }

                $("#q2button").on("click", function() {
                  multiply();
                });
              
                function chessBoard() {
                  var size = prompt("Please enter a number to create a length.");
                  var line = '';

                  var i = 0;

                  for (i = 0; i < size / 2; i++) {
                    line = print + line + '# ';

                    if (i % 2 === 0) {

                    }
                  }

                }

                function countBs(bString) {
                  var count = 0;
                  for (var i = 0; i < bString.length; i++) {
                    if (bString.charAt(i) == 'r') {
                      count = (count + 1);
                    }
                  }
                  return count;
                }


                $("#q4button").on("click", function() {

                  var userInput = prompt("Write a sentance containing the letter 'r' and I will count how many there are.");
                  var x = countBs(userInput);

                  $("#q4result").html(x);
                });
                
                function countChar(cString, charMatch) {
                  var count = 0;
                  for (var i = 0; i < cString.length; i++) {
                    if (cString.charAt(i) == charMatch) {
                      count = (count + 1);
                    }
                  }

                }

                function range(start, end) {
                  var arr = [];
                  if(start <= end){
                    //this is where we establish that the numbers in the range will increase by 1
                    //and start at the start number (2) and end at the end number (6)
                  for( start; start <= end; start++){
                    arr.push(start);
                  }
            }
                  return arr;
                }

                  //2,3,4,5,6
                  $("#q6result").html("The answer is " + rangeAnswer.join());

                        function reverseArray(array) {
                          var output = [];
                          for (var i = array.length - 1; i >= 0; i--){
                            output.push(array[i]);
                        }
                        return output;
                      }

                        $("#q8button").on("click", function() {
                          var output = reverseArray(["A", "B", "C", "D","E"]);
                          $("#q8result").html("The reverse is: " + output);
                    });
                
                function reverseArrayInPlace(array) {
                for (var i = 0; i < Math.floor(array.length / 2); i++) {
                var old = array[i];
                array[i] = array[array.length - 1 - i];
                array[array.length - 1 - i] = old;
                }
                return array;
                }
